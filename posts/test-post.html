  var $element = $('.js-parallax'),
    scrollPos;

  function parallax() {
    if (window.scrollY < $element.height()) {
      scrollPos = Math.round(window.scrollY / 4);
      $element.css({
        transform: 'translateY(' + scrollPos + 'px)'
      });
    }
  }

  window.addEventListener('scroll', function () {
    window.requestAnimFrame(parallax);
  }, false);

  window.requestAnimFrame(parallax);

  parallax();
